IQ Test (6 kyu)

       
function iqTest(numbers) {
    let arr = numbers.split(' '),
        count = 0;

    arr.map(e => (e % 2) ? --count : ++ count);

    if (count > 0) {
        for (let i = 0; i < arr.length; i++) {
            if (arr[i] % 2) {
                return ++i;
            }
        }
    } else {
        for (let i = 0; i < arr.length; i++) {
            if (!(arr[i] % 2)) {
                return ++i;
            }
        }
    }
}
