      function sumStrings(a,b) {
       let result = [];
       let arr1 = a.split('');
       let arr2 = b.split('');
       let min = arr1.length - arr2.length;
       if (min < 0) {
        min *= -1;
        for (let i = 0; i < min; i++) {
         arr1.unshift(0);
        }
       } else if (min > 0) {
        for (let i = 0; i < min; i++) {
         arr2.unshift(0);
        }
       }
       for (let i = 0; i < arr1.length; i++) {
        arr1[i] = Number(arr1[i]);
        arr2[i] = Number(arr2[i]);
       }
       arr1.reverse();
       arr2.reverse();
       for (let i = 0; i < arr1.length; i++) {
        if (arr1[i] + arr2[i] > 9) {
         result.push(arr1[i] + arr2[i] - 10);
         if (i == arr1.length - 1) {
          result.push(1);
         } else {
          arr1[i+1] += 1;
         }
        } else {
         result.push(arr1[i] + arr2[i]);
        }
       }
       result.reverse();
       if (result[0] == 0) {
        result.splice(0, 1);
       }
       return result.join('');
      }
